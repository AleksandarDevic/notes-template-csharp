########################################
# Podman/Docker - Standalone Container (notes-template-app)
########################################

# 1. Generate dev certificate
dotnet dev-certs https -ep "$env:USERPROFILE\.aspnet\https\aspnetapp.pfx" -p root.123
dotnet dev-certs https --trust

# 2. Build image
podman build -t notes-template-app -f src/Web.Api/Dockerfile .
podman image ls notes-template-app

# 3. Create and run container (one-liner)
podman run -p 5001:5001 -p 5000:5000 --name notes-template-app -e ASPNETCORE_ENVIRONMENT=Development -e ConnectionStrings__Database="Host=host.containers.internal;Port=5432;Database=notes_template_db;Username=postgres;Password=root.123;Include Error Detail=true" -e ASPNETCORE_Kestrel__Certificates__Default__Path=/https/aspnetapp.pfx -e ASPNETCORE_Kestrel__Certificates__Default__Password=root.123 -v "$env:USERPROFILE\.aspnet\https:/https:ro" notes-template-app

# 3. Create and run container (multi-line). The ` is PowerShell's line continuation character (equivalent to \ in bash).
podman run -p 5001:8081 -p 5000:8080 --name notes-template-app `
    -e ASPNETCORE_ENVIRONMENT=Development `
    -e ASPNETCORE_HTTPS_PORTS=8081 `
    -e ASPNETCORE_HTTP_PORTS=8080 `
    -e ConnectionStrings__Database="Host=host.containers.internal;Port=5432;Database=notes_template_db;Username=postgres;Password=root.123;Include Error Detail=true" `
    -e ASPNETCORE_Kestrel__Certificates__Default__Path=/https/aspnetapp.pfx `
    -e ASPNETCORE_Kestrel__Certificates__Default__Password=root.123 `
    -v "$env:USERPROFILE\.aspnet\https:/https:ro" `
    notes-template-app

# 4. Stop / Start (pause and resume)
podman stop notes-template-app
podman start notes-template-app

# 5. Remove container (must stop first, or use -f)
podman rm notes-template-app
podman rm -f notes-template-app

# 6. Remove image
podman rmi notes-template-app

# 7. Inspect / Debug
podman inspect notes-template-app
podman exec notes-template-app ls -la /https
podman exec notes-template-app ls -la /app
podman logs -f notes-template-app

########################################
# Podman/Docker Compose (notes-template-api)
########################################

# 1. Build and start (first time or after code changes)
podman compose up --build
podman compose up -d --build

# 2. Stop / Start (pause and resume, containers kept)
podman compose stop
podman compose start

# 3. Down (done for the day, removes containers, keeps volumes)
podman compose down

# 4. Down + volumes (full reset, wipes database)
podman compose down -v

# 5. Rebuild without cache
podman compose build --no-cache

# 6. Restart specific service
podman compose restart webapp

# 7. Logs
podman compose logs -f webapp

# 8. Inspect / Debug
podman inspect notes-template-api
podman exec notes-template-api ls -la /https
podman exec notes-template-api ls -la /app
podman volume ls
podman volume inspect notes_template_db_data
