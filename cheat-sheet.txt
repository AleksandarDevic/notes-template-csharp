########################################
# Podman/Docker - Standalone Container (notes-template-app)
########################################

# 1. Generate dev certificate
dotnet dev-certs https -ep "$env:USERPROFILE\.aspnet\https\aspnetapp.pfx" -p root.123
dotnet dev-certs https --trust

# 2. Build image
podman build -t notes-template-app -f src/Web.Api/Dockerfile .
podman image ls notes-template-app

# 3. Create and run container (one-liner)
podman run -p 5001:5001 -p 5000:5000 --name notes-template-app -e ASPNETCORE_ENVIRONMENT=Development -e ConnectionStrings__Database="Host=host.containers.internal;Port=5432;Database=notes_template_db;Username=postgres;Password=root.123;Include Error Detail=true" -e ASPNETCORE_Kestrel__Certificates__Default__Path=/https/aspnetapp.pfx -e ASPNETCORE_Kestrel__Certificates__Default__Password=root.123 -v "$env:USERPROFILE\.aspnet\https:/https:ro" notes-template-app

# 3. Create and run container (multi-line). The ` is PowerShell's line continuation character (equivalent to \ in bash).
podman run -p 5001:8081 -p 5000:8080 --name notes-template-app `
    -e ASPNETCORE_ENVIRONMENT=Development `
    -e ASPNETCORE_HTTPS_PORTS=8081 `
    -e ASPNETCORE_HTTP_PORTS=8080 `
    -e ConnectionStrings__Database="Host=host.containers.internal;Port=5432;Database=notes_template_db;Username=postgres;Password=root.123;Include Error Detail=true" `
    -e ASPNETCORE_Kestrel__Certificates__Default__Path=/https/aspnetapp.pfx `
    -e ASPNETCORE_Kestrel__Certificates__Default__Password=root.123 `
    -v "$env:USERPROFILE\.aspnet\https:/https:ro" `
    notes-template-app

# 4. Stop / Start (pause and resume)
podman stop notes-template-app
podman start notes-template-app

# 5. Remove container (must stop first, or use -f)
podman rm notes-template-app
podman rm -f notes-template-app

# 6. Remove image
podman rmi notes-template-app

# 7. Inspect / Debug
podman inspect notes-template-app
podman exec notes-template-app ls -la /https
podman exec notes-template-app ls -la /app
podman logs -f notes-template-app

########################################
# Podman/Docker Compose (notes-template-api)
########################################

# 1. Build and start (first time or after code changes)
podman compose up --build
podman compose up -d --build

# 2. Stop / Start (pause and resume, containers kept)
podman compose stop
podman compose start

# 3. Down (done for the day, removes containers, keeps volumes)
podman compose down

# 4. Down + volumes (full reset, wipes database)
podman compose down -v

# 5. Rebuild without cache
podman compose build --no-cache

# 6. Restart specific service
podman compose restart webapp

# 7. Logs
podman compose logs -f webapp

# 8. Inspect / Debug
podman inspect notes-template-api
podman exec notes-template-api ls -la /https
podman exec notes-template-api ls -la /app
podman volume ls
podman volume inspect notes_template_db_data


########################################
# Kubernetes (Minikube)
########################################

# 1. Start minikube
minikube status
minikube start
minikube profile list

# 2. Build app image inside minikube
#    (Podman on host and minikube don't share images, so build directly in minikube)
& minikube -p minikube podman-env --shell powershell | Invoke-Expression   # only for current terminal session
podman image build -t notes-template-app:latest -f src/Web.Api/Dockerfile .

# 3. Deploy Postgres
kubectl apply -f .\k8s\postgres-configmap.yaml
kubectl apply -f .\k8s\postgres-sealed-secret.yaml
kubectl apply -f .\k8s\postgres-pv.yaml
kubectl apply -f .\k8s\postgres-pvc.yaml
kubectl apply -f .\k8s\postgres-deployment.yaml
kubectl apply -f .\k8s\postgres-service.yaml

# 4. Deploy Web API
kubectl create secret generic webapp-cert --from-file=aspnetapp.pfx=$HOME/.aspnet/https/aspnetapp.pfx
kubectl apply -f .\k8s\webapp-configmap.yaml
kubectl apply -f .\k8s\webapp-sealed-secret.yaml
kubectl apply -f .\k8s\webapp-deployment.yaml
kubectl apply -f .\k8s\webapp-service.yaml

# 5. Verify
kubectl get all
kubectl get pods
kubectl get svc
kubectl get pv
kubectl get pvc
kubectl get secret
kubectl get configmap

# 6. Logs / Debug
kubectl logs -f deployment/webapp-deployment
kubectl logs -f deployment/postgres-deployment
kubectl describe pod <pod-name>
kubectl exec -it <pod-name> -- bash
kubectl exec -it <pod-name> -- psql -h localhost -U postgres --password -p 5432
kubectl exec -it <pod-name> -- ls -la /https

# 7. Access services
minikube service list
minikube service webapp-service
minikube service webapp-service --url
kubectl port-forward service/webapp-service 5001:5001 5000:5000    # webapp on localhost:5001/5000
kubectl port-forward deployment/postgres-deployment 5432:5432       # DB on localhost:5432 (e.g. DBeaver)

# 8. After code changes (rebuild + redeploy)
podman image build -t notes-template-app:latest -f src/Web.Api/Dockerfile .
kubectl rollout restart deployment/webapp-deployment

# 9. Teardown
kubectl delete -f .\k8s

########################################
# Sealed Secrets (encrypt secrets for git)
########################################

# 1. Install controller in cluster (one time)
kubectl apply -f https://github.com/bitnami-labs/sealed-secrets/releases/latest/download/controller.yaml
kubectl get pods -n kube-system -l name=sealed-secrets-controller

# 2. Install kubeseal CLI (one time)
#    Download from: https://github.com/bitnami-labs/sealed-secrets/releases/latest
#    Extract kubeseal.exe and copy to C:\Windows\System32\
kubeseal --version

# 3. Verify connection to controller
kubeseal --fetch-cert

# 4. Encrypt secrets (run after changing secret YAML files)
Get-Content .\k8s\postgres-secret.yaml | kubeseal --format yaml > .\k8s\postgres-sealed-secret.yaml
Get-Content .\k8s\webapp-secret.yaml | kubeseal --format yaml > .\k8s\webapp-sealed-secret.yaml

# 5. Apply sealed secrets and restart deployments (pods don't auto-reload env vars)
kubectl apply -f .\k8s\postgres-sealed-secret.yaml
kubectl apply -f .\k8s\webapp-sealed-secret.yaml
kubectl rollout restart deployment/postgres-deployment
kubectl rollout restart deployment/webapp-deployment

# Note: original secret YAML files are in .gitignore (contain plain text passwords)
#       sealed secret YAML files are safe to commit (encrypted with cluster's public key)
